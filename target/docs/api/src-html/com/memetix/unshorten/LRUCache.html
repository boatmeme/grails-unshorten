002    package com.memetix.unshorten;
003    
004    import java.util.Collections;
005    import java.util.Map;
006    import org.apache.commons.collections15.map.LRUMap;
007    
008    /**
009     * LRUCache
010     * 
011     * This class encapsulates the LRUMap from Apache Commons Collections - Generics (http://sourceforge.net/projects/collections/files/)
012     * within a Thread-Safe Wrapper
013     * 
014     * @author Jonathan Griggs  <twitcaps.developer @ gmail.com>
015     * @version     1.0     2011.05.17                              
016     * @since       1.0     2011.05.17    
017     */
018    public class LRUCache {  
019         private Map<Object,Object> map;
020         
021         public LRUCache(int maxSize) {
022             //Defaults to Scan Until Removable when LRUMap is Full
023             map = Collections.synchronizedMap(new LRUMap<Object,Object>(maxSize,true));
024         }
025         
026         public LRUCache(int maxSize,boolean scanUntilRemovable) {
027             map = Collections.synchronizedMap(new LRUMap<Object,Object>(maxSize,scanUntilRemovable));
028         }
029         
030         public int size() {
031             return map.size();
032         }
033         
034         public void put(Object key, Object value) {
035             map.put(key,value);
036         }
037         
038         public Object get(Object key) {
039             return map.get(key);
040         }
041         
042         public void flush() {
043             map.clear();
044         }
045    }
